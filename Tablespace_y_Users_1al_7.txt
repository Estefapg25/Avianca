

--2.	Creación tablespaces

CREATE TABLESPACE AVIANCA
datafile 'Avianca.dbf' size 2000M
EXTENT MANAGEMENT LOCAL
SEGMENT SPACE MANAGEMENT AUTO
ONLINE;

--Creación undo tablespace
CREATE UNDO TABLESPACE  AEROLINEA
datafile 'Aerolinea.dbf' size 25M
RETENTION GUARANTEE;

--3.	Set Undo

ALTER TABLESPACE AEROLINEA DATAFILE ONLINE;

--4.	Creación usuario dba

CREATE USER dba_Avianca 
IDENTIFIED BY 1234
DEFAULT TABLESPACE AVIANCA
QUOTA UNLIMITED ON AVIANCA;

GRANT DBA, CONNECT TO dba_Avianca;

--5.	Creación Perfiles

CREATE PROFILE Empleado LIMIT
SESSIONS_PER_USER 1
IDLE_TIME 10
FAILED_LOGIN_ATTEMPTS 4
PASSWORD_LIFE_TIME 40;

CREATE PROFILE Desarrollador LIMIT
SESSIONS_PER_USER 2
IDLE_TIME 30
PASSWORD_LIFE_TIME 100;

--6.	Creación Usuarios

CREATE USER Zoraida
IDENTIFIED BY 1234
DEFAULT TABLESPACE AVIANCA
PROFILE Empleado;

CREATE USER Gabriel
IDENTIFIED BY 1234
DEFAULT TABLESPACE AVIANCA
PROFILE Empleado;

CREATE USER Hernan
IDENTIFIED BY 1234
DEFAULT TABLESPACE AVIANCA
PROFILE Desarrollador;

CREATE USER Estefania
IDENTIFIED BY 1234
DEFAULT TABLESPACE AVIANCA
